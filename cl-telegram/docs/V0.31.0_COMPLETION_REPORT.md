## 七、使用示例

### 7.1 文件上传

```lisp
;; 创建上传会话
(let ((session (cl-telegram/api:make-upload-session "/path/to/file.zip"
                                                     :chunk-size 524288
                                                     :callback (lambda (&key progress)
                                                                 (format t "进度：~A%~%" progress)))))
  ;; 上传分块
  (with-open-file (stream (cl-telegram/api:upload-session-file-path session)
                          :element-type '(unsigned-byte 8))
    (loop for chunk-index from 0
          for buffer = (make-array 524288 :element-type '(unsigned-byte 8))
          for bytes-read = (read-sequence buffer stream)
          while (plusp bytes-read)
          do (progn
               (when (< bytes-read 524288)
                 (setf buffer (subseq buffer 0 bytes-read)))
               (cl-telegram/api:upload-file-chunk (cl-telegram/api:upload-session-file-id session)
                                                  chunk-index
                                                  buffer)))

  ;; 完成上传
  (cl-telegram/api:complete-upload-session (cl-telegram/api:upload-session-file-id session))))
```

### 7.2 文件下载续传

```lisp
;; 创建下载会话
(let ((session (cl-telegram/api:make-download-session "file_id" "/tmp/file.zip")))
  ;; 下载分块
  (loop for offset from 0 by 524288
        while (< offset file-size)
        do (cl-telegram/api:download-file-chunk (cl-telegram/api:download-session-file-id session)
                                                offset
                                                524288))

  ;; 暂停下载
  (cl-telegram/api:pause-download (cl-telegram/api:download-session-file-id session))

  ;; ... 稍后恢复
  (cl-telegram/api:resume-download (cl-telegram/api:download-session-file-id session)))
```

    ;; 使用库生成 QR 码图片...

  ;; 等待用户扫描
  (loop
     for status = (cl-telegram/api:check-qr-login-status (cl-telegram/api:qr-token-id token))
